JS Day 37: var, let and const

2020-07-30
JavaScript
  • JSnewbie
  • JavaScript
有 N 人看过

var & let

{
  var a = 'A';
  let b = 'B';
}
console.log('var a: ' + a);
console.log('let b: ' + b);

// output
// => "var a: A"
// => ReferenceError: b is not defined

用 var 來宣告一個全域變數，let 的作用域僅 scope 限於 { ... } 區塊。

哪裡適合用 let

for (let i = 0; i < 5; i++) {
    console.log(i);
};

console.log(i);

// output
// 1
// 2
// 3
// 4
// ReferenceError: i is not defined

let i 在 for loop 裡頭才有計次效用，在 { ... } 外呼叫會得到 ReferenceError 錯誤， i 並沒有被宣告。

for (var i = 0; i < 5; i++) {
    console.log(i);
};

console.log(i);

// output
// --- LOOP 開始
// 0
// 1
// 2
// 3
// 4
// --- LOOP 結束
// 5   { ... } 外面呼叫 i 得到的結果

Hoisting 提升

Var Hoisting 變數提升

console.log(a); // undefined
var a = 'A';

// ---- 分隔線 ----

// 實際上這段程式碼對編譯器來說是長這樣
var a; // 只有宣告 a 這個變數
console.log(a); // 只有被宣告，但沒有賦予值，自然是 undefined
var a = 'A';
console.log(a); //=> 'A'

這樣子比較好理解。

var a;
console.log(a);  // undefined

使用 var 跟 let 的結果：

console.log(a); // Undefined
var a = 'A';

console.log(b);  // ReferenceError
let b = 'B';

使用 var 會出現 hoisting，反之 let 則不會出現 hoisting，因為變數 b 之前不存在，所以會出錯。

alpha(); // 'Alpha calling...'
console.log(a); // undefined
var a = 'A';

function alpha() {
    console.log('Alpha calling...');
};

https://drive.google.com/uc?export=view&id=1Pi0OoGeR92Ev1s3pd6uK8I22ovHGjRAQ

var findProd = function() {
  console.log('Product Name');
};
findProd(); // 呼叫函式

正常，沒有出現錯誤

https://drive.google.com/uc?export=view&id=1o6jbz45kT73tefN00vyJBWhXcSgQE1kX

findProd(); // 呼叫函式
var findProd = function() {
  console.log('Product Name');
};

出錯了，findProd is not a function。

https://drive.google.com/uc?export=view&id=1t_2ZwQBRVWG2Bz-sPbRjUSBeqX7AAMXm

這段程式碼編譯器解讀為：

var findProd; // 宣告全域變數
findProd(); // 呼叫 findProduct() 函式
findProd = function() {
  console.log('Product Name');
};

用 let 宣告 函式表達式 function expression

findProd(); // 呼叫函式
let findProd = function() {
  console.log('Product Name');
};

https://drive.google.com/uc?export=view&id=1TG3yEMxGQJ2Y39n2l4vJjl-eJKpxwc0M

用 let 或 const 宣告 function expression 都會出現很明確的錯誤：ReferenceError: can’t access lexical declaration ‘findProd’ before initialization （Temporal Dead Zone, TDZ）。

函式提升 Function Hoisting

console.log(a);
var a;
function a() {
    console.log('say a word');
};

function a(){};
var a;
console.log(a);
function a() {
    console.log('say a word');
};

重複宣告 var, let, const

var a = 'A';
var a = 'B';
console.log(a);
// B

var 可以重複宣告

let a = 'A';
let a = 'B';
console.log(a);
// SyntaxError: redeclaration of let a

let 無法重新宣告

const a = 'A';
const a = 'B';
console.log(a);
// SyntaxError: redeclaration of const a

const 無法重新宣告
